## 파일 열기 / 닫기

```c
#include <stdio.h>

FILE *fp = fopen("data.txt", "r");  // 읽기 모드
if (fp == NULL) {
    perror("파일 열기 실패");
    return -1;
}

// 사용

fclose(fp);
```

### 모드 종류

| 모드 | 설명 |
|---|---|
| `"r"` | 읽기 (파일 없으면 실패) |
| `"w"` | 쓰기 (없으면 생성, 있으면 덮어씀) |
| `"a"` | 추가 (없으면 생성, 있으면 끝에 추가) |
| `"rb"` / `"wb"` | 바이너리 읽기 / 쓰기 |
| `"r+"` | 읽기 + 쓰기 |

## 텍스트 파일 입출력

```c
// 쓰기
FILE *fp = fopen("log.txt", "w");
fprintf(fp, "온도: %.1f\n", 25.3f);
fprintf(fp, "습도: %d%%\n", 60);
fclose(fp);

// 읽기 (한 줄씩)
fp = fopen("log.txt", "r");
char line[128];
while (fgets(line, sizeof(line), fp) != NULL) {
    printf("%s", line);
}
fclose(fp);
```

## 바이너리 파일 입출력

센서 로그, 설정값 저장에 유용합니다:

```c
typedef struct {
    uint32_t timestamp;
    float    temperature;
    float    humidity;
} SensorLog;

// 바이너리 쓰기
SensorLog logs[3] = {
    {1000, 25.3f, 60.1f},
    {2000, 26.0f, 58.5f},
    {3000, 24.8f, 62.3f}
};

FILE *fp = fopen("sensor.bin", "wb");
fwrite(logs, sizeof(SensorLog), 3, fp);
fclose(fp);

// 바이너리 읽기
SensorLog read_buf[3];
fp = fopen("sensor.bin", "rb");
int count = fread(read_buf, sizeof(SensorLog), 3, fp);
fclose(fp);

for (int i = 0; i < count; i++) {
    printf("[%u] 온도: %.1f, 습도: %.1f\n",
            read_buf[i].timestamp,
            read_buf[i].temperature,
            read_buf[i].humidity);
}
```

## 파일 위치 제어

```c
FILE *fp = fopen("data.bin", "rb");

fseek(fp, 0, SEEK_END);        // 끝으로 이동
long size = ftell(fp);          // 현재 위치(= 파일 크기)
fseek(fp, 0, SEEK_SET);        // 처음으로 이동
rewind(fp);                     // 처음으로 이동 (fseek 대안)

fseek(fp, sizeof(SensorLog), SEEK_SET);  // 두 번째 레코드로
SensorLog log;
fread(&log, sizeof(SensorLog), 1, fp);
fclose(fp);
```

## CSV 파일 파싱

```c
fp = fopen("data.csv", "r");
char line[256];

// 헤더 건너뜀
fgets(line, sizeof(line), fp);

while (fgets(line, sizeof(line), fp) != NULL) {
    int   id;
    float val;
    char  name[32];

    // "1,25.3,sensor_A" 형식 파싱
    if (sscanf(line, "%d,%f,%31s", &id, &val, name) == 3) {
        printf("ID:%d 값:%.1f 이름:%s\n", id, val, name);
    }
}
fclose(fp);
```

## 에러 처리

```c
FILE *fp = fopen("config.bin", "rb");
if (fp == NULL) {
    perror("fopen");  // 시스템 에러 메시지 출력
    return -1;
}

size_t n = fread(buf, sizeof(Config), 1, fp);
if (n != 1) {
    if (feof(fp)) fprintf(stderr, "파일 끝 도달\n");
    if (ferror(fp)) fprintf(stderr, "읽기 오류\n");
}
fclose(fp);
```

---

## 정리

- 텍스트: `fprintf`/`fgets`, 바이너리: `fwrite`/`fread`
- 바이너리 모드(`"rb"`, `"wb"`)는 플랫폼 간 줄바꿈 문제를 방지
- `fclose`를 반드시 호출해야 버퍼가 플러시되고 파일이 닫힘
- `feof`와 `ferror`로 읽기 실패 원인 구분
