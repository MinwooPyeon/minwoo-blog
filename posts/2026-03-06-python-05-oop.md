## 클래스 기본

```python
class Sensor:
    # 클래스 변수 (모든 인스턴스 공유)
    count: int = 0

    def __init__(self, name: str, pin: int):
        # 인스턴스 변수
        self.name  = name
        self.pin   = pin
        self._value = 0.0  # 관례적 private (_)
        Sensor.count += 1

    def read(self) -> float:
        self._value = self._hardware_read()
        return self._value

    def _hardware_read(self) -> float:
        return 25.3  # 실제 하드웨어 읽기 (예시)

    def __repr__(self) -> str:
        return f"Sensor(name={self.name!r}, pin={self.pin})"

    def __str__(self) -> str:
        return f"[{self.name}] {self._value:.1f}"

    def __del__(self):
        Sensor.count -= 1

s = Sensor("DHT22", 4)
print(repr(s))  # Sensor(name='DHT22', pin=4)
print(str(s))   # [DHT22] 0.0
print(Sensor.count)  # 1
```

## 프로퍼티 (Property)

getter/setter를 파이썬답게 구현:

```python
class Temperature:
    def __init__(self, celsius: float = 0.0):
        self._celsius = celsius

    @property
    def celsius(self) -> float:
        return self._celsius

    @celsius.setter
    def celsius(self, value: float):
        if value < -273.15:
            raise ValueError("절대 영도 이하 불가")
        self._celsius = value

    @property
    def fahrenheit(self) -> float:
        return self._celsius * 9 / 5 + 32

t = Temperature(25.0)
print(t.celsius)     # 25.0
print(t.fahrenheit)  # 77.0
t.celsius = 30.0
# t.celsius = -300  # ValueError
```

## 상속과 다형성

```python
class Actuator:
    def __init__(self, name: str):
        self.name = name

    def activate(self):
        raise NotImplementedError

    def __repr__(self):
        return f"{type(self).__name__}({self.name!r})"

class LED(Actuator):
    def __init__(self, name: str, pin: int):
        super().__init__(name)
        self.pin = pin

    def activate(self):
        print(f"LED[{self.pin}] ON")

class Buzzer(Actuator):
    def activate(self):
        print(f"Buzzer BEEP")

# 다형성
devices: list[Actuator] = [LED("status", 13), Buzzer("alarm")]
for d in devices:
    d.activate()  # 각 타입에 맞는 activate() 호출
```

## 특수 메서드 (Dunder)

```python
class Vector:
    def __init__(self, x: float, y: float):
        self.x, self.y = x, y

    def __add__(self, other):       return Vector(self.x + other.x, self.y + other.y)
    def __sub__(self, other):       return Vector(self.x - other.x, self.y - other.y)
    def __mul__(self, scalar):      return Vector(self.x * scalar,  self.y * scalar)
    def __abs__(self):              return (self.x**2 + self.y**2) ** 0.5
    def __eq__(self, other):        return self.x == other.x and self.y == other.y
    def __repr__(self):             return f"Vector({self.x}, {self.y})"

v1 = Vector(3, 4)
v2 = Vector(1, 2)
print(v1 + v2)   # Vector(4, 6)
print(abs(v1))   # 5.0
```

## dataclass (Python 3.7+)

보일러플레이트 자동 생성:

```python
from dataclasses import dataclass, field

@dataclass
class SensorReading:
    name:      str
    value:     float
    unit:      str   = "°C"
    timestamp: float = field(default_factory=lambda: time.time())

    def is_valid(self) -> bool:
        return -40.0 <= self.value <= 125.0

r = SensorReading("DHT22", 25.3)
print(r)  # SensorReading(name='DHT22', value=25.3, unit='°C', ...)
```

## 추상 클래스

```python
from abc import ABC, abstractmethod

class ISerialPort(ABC):
    @abstractmethod
    def open(self, baud: int) -> bool: ...

    @abstractmethod
    def write(self, data: bytes) -> int: ...

    @abstractmethod
    def read(self, size: int) -> bytes: ...

    @abstractmethod
    def close(self): ...

class PySerialPort(ISerialPort):
    def open(self, baud: int) -> bool:
        import serial
        self._ser = serial.Serial(baud=baud)
        return True
```

## 정리

- `_name`: 관례적 private, `__name`: 이름 맹글링(name mangling)
- `@property`로 getter/setter를 속성처럼 사용
- `super().__init__()`으로 부모 생성자 명시적 호출
- `@dataclass`로 단순 데이터 클래스의 보일러플레이트 제거
- `ABC`로 인터페이스 정의
