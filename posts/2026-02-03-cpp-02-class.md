## 클래스 기본 구조

```cpp
class Sensor {
private:
    string name;
    float  value;
    bool   is_active;

public:
    // 생성자
    Sensor(const string &n, float v = 0.0f)
        : name(n), value(v), is_active(false) {}

    // 멤버 함수
    void   activate()          { is_active = true; }
    void   set_value(float v)  { value = v; }
    float  get_value()   const { return value; }
    string get_name()    const { return name; }
    bool   active()      const { return is_active; }

    void print() const {
        cout << "[" << name << "] "
             << (is_active ? "ON" : "OFF")
             << " value=" << value << endl;
    }
};

int main() {
    Sensor temp("DHT22_TEMP", 25.3f);
    temp.activate();
    temp.set_value(26.1f);
    temp.print();  // [DHT22_TEMP] ON value=26.1
}
```

## getter / setter

```cpp
class Temperature {
private:
    float celsius;

public:
    Temperature(float c = 0.0f) : celsius(c) {}

    float get_celsius()    const { return celsius; }
    float get_fahrenheit() const { return celsius * 9.0f / 5.0f + 32.0f; }
    float get_kelvin()     const { return celsius + 273.15f; }

    void set_celsius(float c) {
        if (c < -273.15f) c = -273.15f;  // 절대 영도 이하 방지
        celsius = c;
    }
};

Temperature t(25.0f);
cout << t.get_fahrenheit() << endl;  // 77
cout << t.get_kelvin()     << endl;  // 298.15
```

## this 포인터

멤버 함수 내에서 객체 자신을 가리킵니다:

```cpp
class Counter {
    int count;
public:
    Counter(int c = 0) : count(c) {}

    // 메서드 체이닝 (Fluent Interface)
    Counter &increment() { count++; return *this; }
    Counter &add(int n)  { count += n; return *this; }
    Counter &reset()     { count = 0; return *this; }
    int get() const      { return count; }
};

Counter c;
int val = c.increment().increment().add(5).get();
cout << val << endl;  // 7
```

## static 멤버

모든 객체가 공유하는 멤버:

```cpp
class SensorManager {
    static int instance_count;
    int id;

public:
    SensorManager() {
        id = ++instance_count;
    }
    ~SensorManager() {
        --instance_count;
    }

    static int get_count() { return instance_count; }
    int get_id()     const { return id; }
};

int SensorManager::instance_count = 0;  // 정의 필수

SensorManager s1, s2, s3;
cout << SensorManager::get_count() << endl;  // 3
```

## 클래스 분리 (헤더 / 구현 파일)

```cpp
// motor.h
#pragma once
#include <string>

class Motor {
    int   pin;
    int   speed;
    bool  running;

public:
    Motor(int pin, int max_speed = 255);
    void start(int speed);
    void stop();
    bool is_running() const;
};
```

```cpp
// motor.cpp
#include "motor.h"
#include <iostream>

Motor::Motor(int pin, int max_speed)
    : pin(pin), speed(0), running(false) {}

void Motor::start(int spd) {
    speed   = spd;
    running = true;
    std::cout << "Motor[" << pin << "] start speed=" << speed << "\n";
}

void Motor::stop() {
    speed   = 0;
    running = false;
}

bool Motor::is_running() const { return running; }
```

## 정리

- 데이터는 `private`, 인터페이스는 `public`으로 캡슐화
- `const` 멤버 함수는 객체 상태를 변경하지 않는 함수에 반드시 붙임
- `this` 포인터로 메서드 체이닝 구현
- 헤더(.h)에 선언, 소스(.cpp)에 구현 분리가 C++ 표준 패턴
